CREATE  TABLE IF NOT EXISTS `alumno`.`ALUMNO` (

  `ID_ALUMNO` INT NOT NULL AUTO_INCREMENT COMMENT 'Identificador del alumno' ,

  `NOMBRE_ALUMNO` VARCHAR(120) NOT NULL COMMENT 'Nombre del alumno' ,

  `CLAVE_ALUMNO` VARCHAR(45) NOT NULL COMMENT 'Clave del alumno' ,

  `CALIFICACION_ALUMNO` VARCHAR(5) NOT NULL COMMENT 'Calificacion del alumno' ,

  `FECHA_REGISTRO` TIMESTAMP NOT NULL COMMENT 'Fecha de registro en la tabla del alumno' ,

  PRIMARY KEY (`ID_ALUMNO`) ,

  UNIQUE INDEX `CLAVE_ALUMNO_UNIQUE` (`CLAVE_ALUMNO` ASC) )

ENGINE = InnoDB;
